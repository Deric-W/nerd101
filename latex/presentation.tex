\section{macros}
\begin{frame}[fragile]{macros}
No one whats to type this to create a matrix
    \begin{lstlisting}
    \begin{pmatrix} a & b \\ c & d \end{pmatrix}
    \end{lstlisting}
so we define a macro
    \begin{lstlisting}
\newcommand{\m}[1] {
    \begin{pmatrix}
    #1
    \end{pmatrix}}
    \end{lstlisting}
and now we can type
    \begin{lstlisting}
    \m{a&b\\c&d}
    \end{lstlisting}
\end{frame}

\section{reference management}
\begin{frame}[fragile]{reference management with bibtex}
    \begin{itemize}
        \item it can be hard to keep track of all sources one used
        \item referncing papers, articles or books can be hard
        \item different standarts on referencing are confusing
    \end{itemize}
    \begin{lstlisting}
@inbook{knuth,
    author = "Donald E. Knuth",
    title = "Fundamental Algorithms",
    publisher = "Addison-Wesley",
    year = "1973",
    chapter = "1.2",
    keywords  = "knuth,programming"
}
    \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{including zour bibtex file}
    \begin{lstlisting}
\usepackage{biblatex}
\addbibresource{sample.bib}
    \end{lstlisting}

now you just reference it in your text with

 \begin{lstlisting}
\cite{knuth}
\end{lstlisting}
and print your references at the end of your document
\begin{lstlisting}
\printbibliography
\end{lstlisting}
\end{frame}
